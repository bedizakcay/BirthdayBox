function daysUntil(month,day){

const today=new Date();

let next=new Date(today.getFullYear(),month-1,day);

if(next<today) next=new Date(today.getFullYear()+1,month-1,day);

const diff=Math.ceil((next-today)/(1000*60*60*24));

return diff===365?0:diff;

}

function calcAge(year,month,day){

if(!year) return null;

const today=new Date();

let age=today.getFullYear()-year;

if(today.getMonth()+1<month||(today.getMonth()+1===month&&today.getDate()<day))age--;

return age;

}

function nextAge(year,month,day){

if(!year)return null;

return daysUntil(month,day)===0?calcAge(year,month,day):calcAge(year,month,day)+1;

}



function launchConfetti(){

el.style.width=(Math.random()*10+6)+"px";

el.style.height=(Math.random()*10+6)+"px";

el.style.borderRadius=Math.random()>0.5?"50%":"2px";

document.body.appendChild(el);

setTimeout(()=>el.remove(),4000);

}

}


function playSound(){

try{

const ctx=new(window.AudioContext||window.webkitAudioContext)();

const notes=[523,659,784,1047];

notes.forEach((freq,i)=>{

const o=ctx.createOscillator();

const g=ctx.createGain();


o.connect(g);g.connect(ctx.destination);

o.frequency.value=freq;

o.type="sine";

g.gain.setValueAtTime(0,ctx.currentTime+i*0.15);

g.gain.linearRampToValueAtTime(0.3,ctx.currentTime+i*0.15+0.05);

g.gain.linearRampToValueAtTime(0,ctx.currentTime+i*0.15+0.3);

o.start(ctx.currentTime+i*0.15);

o.stop(ctx.currentTime+i*0.15+0.3);

});

}catch(e){}

}


async function requestNotificationPermission(){

if(!('Notification' in window)) return false;

if(Notification.permission==='granted') return true;

const result=await Notification.requestPermission();

return result==='granted';

}

new Notification("🎂 Happy Birthday!", {body:`Today is ${name}'s birthday! Don't forget to wish them!`, icon:"data:image/svg+xml,<svg xmlns='http://


www.w3.org/2000/svg' viewBox='0 0 100 100'><text y='.9em' font-size='90'>🎂</text></svg>", vibrate:[200,100,200]});

}

}
